CMS API
/api/file 
PARAMETERS:N/A | RETURN:html | METHOD:get | DESC:File browser
/api/file/info/{fileId} 
PARAMETERS:{fileId} | RETURN:JSON | METHOD:get | DESC:retrieve file information
/api/file/search/{keyword} 
PARAMETERS:keyword | RETURN:JSON | METHOD:get | DESC:keyword search
/api/file/lock 
PARAMETERS:{fileId, passcode} | RETURN:JSON | METHOD:post | DESC: lock specific file by given passcode
/api/file/unlock 
PARAMETERS:{fileId, passcode} | RETURN:JSON | METHOD:post | DESC: unlock specific file by given passcode
/api/file/save 
PARAMETERS:{content, title} | RETURN:JSON | METHOD:post | DESC:save text/html content
/api/file/upload 
PARAMETERS:{file, site} | RETURN:JSON | METHOD:post | DESC:upload file
/api/file/stream/{fileId} 
PARAMETERS:{fileId} | RETURN:binary stream | METHOD:get | DESC:get file streaming
/api/file/download/{fileId} 
PARAMETERS:{fileId} | RETURN:binary | METHOD:get | DESC:download file
/api/file/remove/{fileId} 
PARAMETERS:{fileId} | RETURN:binary | METHOD:delete | DESC:delete file
/api/label/search 
PARAMETERS:{prefix} | RETURN:json | METHOD:get | DESC:get all label counts
/api/label/add/{fileId}/{label} 
PARAMETERS:{fileId, label} | RETURN:json | METHOD:post | DESC:add new label to file
/api/label/remove/{fileId}/{label} 
PARAMETERS:{fileId, label} | RETURN:json | METHOD:post | DESC:remove label from file
